#include <iostream>

using namespace std;

main(){

    int i = 0, num[3], maior = 0, menor = 999999999;

    for(i = 0; i < 3; i++){
        cout << "Digite o numero: \n";
        cin >> num[i];

        if(num[i] > maior){
            maior = num[i];
        }

        if(num[i] < menor){
            menor = num[i];
        }

    }
    cout << "Maior: " << maior << "\n";
    cout << "Menor: " << menor << "\n";

    return 0;
}
